add_namespace = belgium

# Flemish nationalists in power: Split-up of Belgium
country_event = {
    id = belgium.1
    title = belgium.1.t
    desc = belgium.1.d
    picture = GFX_report_factory_opened
 
    trigger = {
		BEL = {
			has_government = natsoc
			owns_state = 6
			has_war = no
		}
    }
 
    mean_time_to_happen = {
        days = 3
    }
	
	immediate = {
		set_global_flag = belgium_splitup
	}
	
	option = {
        name = belgium.1.a
		FLA = {
			transfer_state = 6
		}
		FLA = {
			change_tag_from = BEL
		}
		FLA = {
			set_politics = {
				ruling_party = natsoc
				elections_allowed = no
			}
		}
		if = {
			limit = {
				OR = {
					BEL = {
						owns_state = 34
					}
					FLA = {
						owns_state = 34
					}
					HOL = {
						owns_state = 34
					}
				}
			}
			BEL = { release = WAL }
		}
		if = {
			limit = {
				AND = {
					NOT = {
						OR = {
							BEL = {
								owns_state = 34
							}
							FLA = {
								owns_state = 34
							}
							HOL = {
								owns_state = 34
							}
						}
					}
					OR = {
						BEL = {
							owns_state = 768
						}
						FLA = {
							owns_state = 768
						}
						HOL = {
							owns_state = 768
						}
					}
				}
			}
			768 = {
				add_core_of = FLA
				remove_core_of = WAL
			}
		}
    }
}
